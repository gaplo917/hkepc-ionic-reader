# Dependencies

		// Clone Repo
		git clone https://github.com/gaplo917/hkepc-ionic-reader
		
		cd hkepc-ionic-reader
		
		npm install
		
		bower install

		// Use node modules
		npm install browserify -g
		
		// Build modules
		npm install gulp -g


# Run
		// run in web
		gulp run
		
		// build production
		gulp release
